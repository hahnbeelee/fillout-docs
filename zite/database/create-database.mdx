---
title: Create Database
description: Creates a new database with tables and fields.
api: POST /api/v1/bases
---

Creates a new database with tables and fields.

## Headers

<ParamField header="Authorization" type="string" required>
  Enter your Fillout API key. Format: Bearer &lt;api_key&gt;
</ParamField>

## Request Body

<ParamField body="name" type="string" required>
  Database name (minimum 1 character)
</ParamField>

<ParamField body="tables" type="array" required>
  Array of table definitions to create with the database (at least one table required)

  <Expandable title="Table definition structure">
    <ParamField body="name" type="string" required>
      Table name (minimum 1 character)
    </ParamField>
    <ParamField body="fields" type="array" required>
      Array of field definitions to create with the table (at least one field required)
      
      <Expandable title="Field definition structure">
        <ParamField body="type" type="string" required>
          Field type (see [Field Types Reference](/zite/database/field-types))
        </ParamField>
        <ParamField body="name" type="string" required>
          Field name (minimum 1 character)
        </ParamField>
        <ParamField body="template" type="object" required>
          Field-specific configuration options that vary by field type
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="workspaceId" type="number">
  Optional workspace identifier to create the database in a specific workspace
</ParamField>

## Response

Returns the newly created database with generated ID, tables, fields, and views.

<ResponseField name="id" type="string">
  Auto-generated unique identifier for the database (starts with `base_`)
</ResponseField>

<ResponseField name="name" type="string">
  The database name that was specified in the request
</ResponseField>

<ResponseField name="workspaceId" type="number">
  The workspace ID where the database was created
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO timestamp of when the database was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  ISO timestamp of when the database was last updated
</ResponseField>

<ResponseField name="tables" type="array">
  Array of created table objects with their generated IDs, fields, and views
  
  <Expandable title="Table object structure">
    <ResponseField name="id" type="string">
      Auto-generated unique identifier for the table
    </ResponseField>
    <ResponseField name="name" type="string">
      The table name
    </ResponseField>
    <ResponseField name="order" type="number">
      The display order of the table within the database
    </ResponseField>
    <ResponseField name="primaryFieldId" type="string">
      The ID of the first field, which serves as the primary field
    </ResponseField>
    <ResponseField name="fields" type="array">
      Array of created field objects with their generated IDs and configurations
    </ResponseField>
    <ResponseField name="views" type="array">
      Array of default views created for the table
    </ResponseField>
  </Expandable>
</ResponseField>

<Warning>
  - Each table requires at least one field
  - Table names must be unique within your database
  - Field names must be unique within each table
</Warning>
